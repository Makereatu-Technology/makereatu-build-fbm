CREATE TABLE IF NOT EXISTS documents (
    doc_id UUID PRIMARY KEY,
    source_ref TEXT,
    bank TEXT,
    doc_type TEXT,
    project_id TEXT,
    project_name TEXT,
    classification_confidence DOUBLE PRECISION
);

CREATE TABLE IF NOT EXISTS extracted_fields (
    id BIGSERIAL PRIMARY KEY,
    doc_id UUID REFERENCES documents(doc_id),
    field_name TEXT,
    field_value TEXT,
    normalized_value JSONB,
    unit TEXT,
    page INTEGER,
    source_type TEXT,
    evidence_text TEXT,
    confidence DOUBLE PRECISION,
    debug_json JSONB
);

CREATE TABLE IF NOT EXISTS indicators (
    id BIGSERIAL PRIMARY KEY,
    doc_id UUID REFERENCES documents(doc_id),
    indicator_name TEXT,
    indicator_value DOUBLE PRECISION,
    unit TEXT,
    method TEXT,
    provenance_json JSONB
);

CREATE OR REPLACE VIEW vw_project_core AS
SELECT
    d.doc_id,
    d.bank,
    d.doc_type,
    d.project_id,
    d.project_name,
    MAX(CASE WHEN f.field_name = 'planned_closing_date' THEN f.normalized_value::text END) AS planned_closing_date,
    MAX(CASE WHEN f.field_name = 'actual_closing_date' THEN f.normalized_value::text END) AS actual_closing_date,
    MAX(CASE WHEN f.field_name = 'appraisal_cost' THEN f.normalized_value::text END) AS appraisal_cost,
    MAX(CASE WHEN f.field_name = 'actual_cost' THEN f.normalized_value::text END) AS actual_cost,
    MAX(CASE WHEN f.field_name = 'output_target_km' THEN f.normalized_value::text END) AS output_target_km,
    MAX(CASE WHEN f.field_name = 'output_actual_km' THEN f.normalized_value::text END) AS output_actual_km,
    MAX(CASE WHEN f.field_name = 'err_percent' THEN f.normalized_value::text END) AS err_percent,
    MAX(CASE WHEN f.field_name = 'eirr_percent' THEN f.normalized_value::text END) AS eirr_percent
FROM documents d
LEFT JOIN extracted_fields f ON d.doc_id = f.doc_id
GROUP BY d.doc_id, d.bank, d.doc_type, d.project_id, d.project_name;

SELECT
    d.project_name,
    d.bank,
    i1.indicator_value AS delay_years,
    i2.indicator_value AS delivery_gap_km_ratio,
    i3.indicator_value AS economic_rate_percent
FROM documents d
LEFT JOIN indicators i1 ON d.doc_id = i1.doc_id AND i1.indicator_name = 'delay_years'
LEFT JOIN indicators i2 ON d.doc_id = i2.doc_id AND i2.indicator_name = 'delivery_gap_km_ratio'
LEFT JOIN indicators i3 ON d.doc_id = i3.doc_id AND i3.indicator_name = 'economic_rate_percent';

SELECT d.project_name, d.bank, i.indicator_name, i.indicator_value
FROM documents d
LEFT JOIN indicators i
  ON d.doc_id = i.doc_id
ORDER BY d.project_name, i.indicator_name;

SELECT
    d.project_name,
    d.bank,
    MAX(CASE WHEN i.indicator_name = 'delay_years' THEN i.indicator_value END) AS delay_years,
    MAX(CASE WHEN i.indicator_name = 'delivery_gap_km_ratio' THEN i.indicator_value END) AS delivery_gap_km_ratio,
    MAX(CASE WHEN i.indicator_name = 'economic_rate_percent' THEN i.indicator_value END) AS economic_rate_percent
FROM documents d
LEFT JOIN indicators i
  ON d.doc_id = i.doc_id
GROUP BY d.project_name, d.bank
ORDER BY d.project_name;

CREATE OR REPLACE VIEW vw_project_indicators AS
SELECT
    d.doc_id,
    d.project_name,
    d.bank,
    MAX(CASE WHEN i.indicator_name = 'delay_years' THEN i.indicator_value END) AS delay_years,
    MAX(CASE WHEN i.indicator_name = 'delay_months' THEN i.indicator_value END) AS delay_months,
    MAX(CASE WHEN i.indicator_name = 'delivery_gap_km_ratio' THEN i.indicator_value END) AS delivery_gap_km_ratio,
    MAX(CASE WHEN i.indicator_name = 'economic_rate_percent' THEN i.indicator_value END) AS economic_rate_percent
FROM documents d
LEFT JOIN indicators i
  ON d.doc_id = i.doc_id
GROUP BY d.doc_id, d.project_name, d.bank;
